JavaScript ternary operator ? :

syntax : condition ? If expression TRUE : If Expression False

let age = 18;
let message ='';
age >= 16 ? (message ="You can Derive"):(message="You can Not Drive");
console.log(message);
//Output : You can Derive.

Multiple JavaScript ternary operators example

let speed = 90;
let message2 = speed >= 120 ? 'To Fast' : speed >= 80 ? 'Fast' : 'OK';
console.log(message2)
// Output: Fast

JavaScript switch case statement

let day = 3;
let dayname;
switch (day) {
case 1 :
dayname = 'Sunday';
break;
case 2:
dayname = 'Monday';
break;
case 3:
dayname = 'Tuesday';
break;
case 4:
dayname = 'Wednesday';
break;
case 5:
dayname = 'Thursday';
break;
case 6:
dayname = 'Friday';
break;
case 7:
dayname = 'Saturday';
break;
default:
dayname = 'Invalid Day';
}
document.writeln(`Today is :${dayname}`);
